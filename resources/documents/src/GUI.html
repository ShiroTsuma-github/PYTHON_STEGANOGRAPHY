 19def get_config_path() -> str:
 20    """Return valid path for app, when run in main.py or exe,
 21    to allow for permanent settings.
 22
 23    Returns:
 24        str: Valid path to use
 25    """
 26    if hasattr(sys, "_MEIPASS"):
 27        abs_home = os.path.abspath(os.path.expanduser("~"))
 28        abs_dir_app = os.path.join(abs_home, ".Steganograph")
 29        if not os.path.exists(abs_dir_app):
 30            os.mkdir(abs_dir_app)
 31        cfg_path = os.path.join(abs_dir_app, "config.json")
 32    else:

 34    return cfg_path

 38    """Loads config.json, that stores default path to images.
 39
 40    Args:

 42
 43    Returns:

 58def save_json(option: str, image_directory: str) -> None:
 59    """Saves setting to json. Based on main.py or .exe location is different.
 60
 61    Args:
 62        option (str): Option to overwrite or add
 63        image_directory (str): Location to add
 64    """
 65    path = get_config_path()
 66    data = {option: image_directory}
 67    if os.path.exists(path):
 68        with open(path, "r+") as json_file:
 69            data = json.load(json_file)
 70        data.update({option: image_directory})
 71    with open(path, "w") as config_file:
 72        json.dump(data, config_file)
